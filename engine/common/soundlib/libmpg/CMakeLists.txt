cmake_minimum_required(VERSION 2.6.0)
project (XASH_LIBMPG)

# set (CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -pedantic")
set (CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -w")
set (CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -g")
set (CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -O3")
set (CMAKE_CXX_FLAGS_MINSIZEREL "${CMAKE_CXX_FLAGS_MINSIZEREL} -Os")
set (CMAKE_CXX_FLAGS_RELWITHDEBINFO "${CMAKE_CXX_FLAGS_RELWITHDEBINFO} -g -O3")

set (XASH_LIBMPG_LIBRARY xashmpg)

file (GLOB XASH_LIBMPG_SOURCES *.c *.h)

if(PANDORA OR ODROID)
 file (GLOB XASH_LIBMPG_ASM *.S)
endif()

add_library (${XASH_LIBMPG_LIBRARY} STATIC ${XASH_LIBMPG_SOURCES} ${XASH_LIBMPG_ASM})

set_target_properties (${XASH_LIBMPG_SHARED} PROPERTIES
	VERSION ${XASH3D_VERSION} SOVERSION ${XASH3D_VERSION}
	POSITION_INDEPENDENT_CODE 1)
